	function replacePicUrl($content = null, $strUrl = null) {
		if ($strUrl) {

	    	preg_match_all("/<img(.*)src=\"([^\"]+)\"(.*)[^>]+>/isU",$content,$matches);
	/**

	 */
	    	$img = "";  
	        if(!empty($matches)) {
